bipedal_ultra_speed.c

Demonstracja ultra speed mode:

#include "bipedal_gait.h"
void ultra_speed_demo() {
// Sprawdzenie aktualnej konfiguracji
printBipedalConfig();
// Ultra speed sprint
printf("=== ULTRA SPEED SPRINT ===\n");
uint32_t start_time = HAL_GetTick();
bipedalGaitWalk(&pca1, &pca2, BIPEDAL_FORWARD, 5);
uint32_t total_time = HAL_GetTick() - start_time;
printf("5 cykli w %lu ms (%.1f cm/s)\n",
total_time, (5 * 4.0f * 1000.0f) / total_time);
}
