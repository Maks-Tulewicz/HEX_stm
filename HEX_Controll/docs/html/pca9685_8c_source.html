22bool PCA9685_Init(PCA9685_Handle_t *handle, I2C_HandleTypeDef *hi2c, uint8_t address)
23{
24 // Validate input parameters
25 if (handle == NULL || hi2c == NULL)
26 {
27 return false;
28 }
29
30 // Initialize handle structure
31 handle->hi2c = hi2c;
32 handle->address = address;
33 handle->ready = false;
34
35 // Test I2C communication first
36 if (HAL_I2C_IsDeviceReady(hi2c, address << 1, 3, 1000) != HAL_OK)
37 {
38 return false;
39 }
40
41 // Step 1: Set normal mode with auto-increment (WORKING VALUE: 0x20)
42 uint8_t mode1_data = 0x20; // Auto-increment enabled (bit 5)
43 if (HAL_I2C_Mem_Write(hi2c, address << 1, PCA9685_MODE1, 1, &mode1_data, 1, 1000) != HAL_OK)
44 {
45 return false;
46 }
47
48 // Step 2: Set frequency to 50Hz (WORKING PRESCALER: 121)
49 // Formula: prescaler = round(25MHz/(4096*50Hz)) - 1 = 121
50 uint8_t prescaler = 121;
51
52 // Enter sleep mode to change prescaler
53 uint8_t sleep_mode = 0x10; // Sleep bit set (bit 4)
54 if (HAL_I2C_Mem_Write(hi2c, address << 1, PCA9685_MODE1, 1, &sleep_mode, 1, 1000) != HAL_OK)
55 {
56 return false;
57 }
58
59 // Write prescaler register
60 if (HAL_I2C_Mem_Write(hi2c, address << 1, PCA9685_PRESCALE, 1, &prescaler, 1, 1000) != HAL_OK)
61 {
62 return false;
63 }
64
65 // Exit sleep mode (restore normal mode)
66 if (HAL_I2C_Mem_Write(hi2c, address << 1, PCA9685_MODE1, 1, &mode1_data, 1, 1000) != HAL_OK)
67 {
68 return false;
69 }
70
71 // Small delay for oscillator to stabilize
72 HAL_Delay(5);
73
74 handle->ready = true;
75 return true;
76}

86bool PCA9685_SetServoAngle(PCA9685_Handle_t *handle, uint8_t channel, float angle)
87{
88 if (handle == NULL || !handle->ready || channel > 15)
89 {
90 return false;
91 }
92
93 // Limit angle to 0-180° range
94 if (angle < 0)
95 angle = 0;
96 if (angle > 180)
97 angle = 180;
98
99 // Linear interpolation for full 180° range
100 uint16_t pwm_range = SERVO_PWM_MAX - SERVO_PWM_MIN; // 650 - 100 = 550
101 uint16_t pwm_value = SERVO_PWM_MIN + (uint16_t)((angle / 180.0f) * pwm_range);
102
103 return PCA9685_SetPWM(handle, channel, pwm_value);
104}

143bool PCA9685_SetPWM(PCA9685_Handle_t *handle, uint8_t channel, uint16_t pwm_value)
144{
145 if (handle == NULL || !handle->ready || channel > 15)
146 {
147 return false;
148 }
149
150 // Limit PWM to 12-bit maximum
151 if (pwm_value > 4095)
152 {
153 pwm_value = 4095;
154 }
155
156 // Calculate base register for this channel
157 // Channel 0 = 0x06, Channel 1 = 0x0A, etc. (each channel uses 4 registers)
158 uint8_t base_reg = PCA9685_LED0_ON_L + (4 * channel);
159
160 // Prepare PWM data exactly like the working manual commands:
161 // [ON_L, ON_H, OFF_L, OFF_H] = [0, 0, pwm_low, pwm_high]
162 uint8_t pwm_data[4] = {
163 0x00, // ON_L: Low byte of ON time (0)
164 0x00, // ON_H: High byte of ON time (0)
165 pwm_value & 0xFF, // OFF_L: Low byte of OFF time
166 (pwm_value >> 8) & 0xFF // OFF_H: High byte of OFF time
167 };
168
169 // Write all 4 registers in one transaction (auto-increment enabled)
170 // This replicates: HAL_I2C_Mem_Write(&hi2c1, address<<1, base_reg, 1, pwm_data, 4, 1000)
171 if (HAL_I2C_Mem_Write(handle->hi2c, handle->address << 1, base_reg, 1, pwm_data, 4, 1000) != HAL_OK)
172 {
173 return false;
174 }
175
176 return true;
177}
