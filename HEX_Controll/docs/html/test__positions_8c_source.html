8// Funkcja do ustawienia kątów stawów nogi na serwa
9void setLegJoints(PCA9685_Handle_t *pca, int leg_number, float q1, float q2, float q3)
10{
11 // Mapowanie kanałów dla nóg (I2C1 - nogi lewe)
12 uint8_t base_channel;
13 if (leg_number == 1)
14 base_channel = 0; // Noga 1: kanały 0,1,2
15 else if (leg_number == 3)
16 base_channel = 3; // Noga 3: kanały 3,4,5
17 else if (leg_number == 5)
18 base_channel = 6; // Noga 5: kanały 6,7,8
19 else
20 {
21 printf("Noga %d nie jest obsługiwana (tylko 1,3,5)\n", leg_number);
22 return;
23 }
24
25 // Konwersja radianów na stopnie
26 float hip_deg = q1 * 180.0f / M_PI;
27 float knee_deg = q2 * 180.0f / M_PI;
28 float ankle_deg = q3 * 180.0f / M_PI;
29
30 // TYMCZASOWE mapowanie - do kalibracji
31 // Zakładamy że 0 rad = 90° servo (pozycja neutralna)
32 float servo_hip = 90.0f + hip_deg;
33 float servo_knee = 90.0f + knee_deg;
34 float servo_ankle = 90.0f + ankle_deg;
35
36 // Ograniczenia serw
37 if (servo_hip < 0.0f)
38 servo_hip = 0.0f;
39 if (servo_hip > 180.0f)
40 servo_hip = 180.0f;
41 if (servo_knee < 0.0f)
42 servo_knee = 0.0f;
43 if (servo_knee > 180.0f)
44 servo_knee = 180.0f;
45 if (servo_ankle < 0.0f)
46 servo_ankle = 0.0f;
47 if (servo_ankle > 180.0f)
48 servo_ankle = 180.0f;
49
50 printf("Noga %d - IK angles [deg]: hip=%.1f, knee=%.1f, ankle=%.1f\n",
51 leg_number, hip_deg, knee_deg, ankle_deg);
52 printf("Noga %d - Servo angles [deg]: hip=%.1f, knee=%.1f, ankle=%.1f\n",
53 leg_number, servo_hip, servo_knee, servo_ankle);
54 printf("Noga %d - Kanały: %d, %d, %d\n",
55 leg_number, base_channel, base_channel + 1, base_channel + 2);
56
57 // Ustaw serwa
58 PCA9685_SetServoAngle(pca, base_channel + 0, servo_hip);
59 PCA9685_SetServoAngle(pca, base_channel + 1, servo_knee);
60 PCA9685_SetServoAngle(pca, base_channel + 2, servo_ankle);
61}

181bool legIKToServos(int leg_number, float x, float y, float z, PCA9685_Handle_t *pca)
182{
183 float q1, q2, q3;
184 bool success = computeLegIK(leg_number, x, y, z, &q1, &q2, &q3);
185
186 if (success)
187 {
188 setLegJoints(pca, leg_number, q1, q2, q3);
189 printf("Pozycja (%.1f, %.1f, %.1f) ustawiona dla nogi %d\n", x, y, z, leg_number);
190 }
191 else
192 {
193 printf("IK failed dla pozycji (%.1f, %.1f, %.1f) nogi %d\n", x, y, z, leg_number);
194 }
195

197}
