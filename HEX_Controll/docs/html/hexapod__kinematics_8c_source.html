18bool computeLegIK(int leg_number, float x, float y, float z,
19 float *q1, float *q2, float *q3)
20{
21 if (leg_number < 1 || leg_number > 6 || q1 == NULL || q2 == NULL || q3 == NULL)
22 {
23 return false;
24 }
25
26 // Pobierz konfigurację dla danej nogi
27 const LegOrigin_t *leg = &leg_origins[leg_number - 1];
28
29 printf("Leg %d IK input - x: %.2f, y: %.2f, z: %.2f\n", leg_number, x, y, z);
30
31 // 1. Przekształcenie do lokalnego układu współrzędnych nogi
32 float local_x = x - leg->x;
33 float local_y = y - leg->y;
34
35 printf("Leg %d - local coords: x=%.3f, y=%.3f\n", leg_number, local_x, local_y);
36
37 // 2. Obliczenie kąta biodra (obrót wokół osi Z)
38 *q1 = atan2f(local_y, local_x);
39
40 // Inwersja kąta biodra dla prawych nóg
41 if (leg->invert_hip)
42 {
43 if (*q1 > 0)
44 *q1 = *q1 - M_PI;
45 else
46 *q1 = *q1 + M_PI;
47 }
48
49 printf("Leg %d - hip angle before constraints: %.2f deg\n", leg_number, *q1 * 180.0f / M_PI);
50
51 // 3. Obliczenie odległości radialnej od osi biodra
52 float r = sqrtf(local_x * local_x + local_y * local_y) - L1;
53 float h = -z; // Zmiana znaku, bo oś Z jest skierowana w dół
54
55 printf("Leg %d - r=%.2f, h=%.2f\n", leg_number, r, h);
56
57 // 4. Sprawdzenie czy punkt jest w zasięgu nogi
58 float D2 = r * r + h * h;
59 float D = sqrtf(D2);
60
61 printf("Leg %d - distance D=%.2f, max_reach=%.2f, min_reach=%.2f\n",
62 leg_number, D, L2 + L3, fabsf(L2 - L3));
63
64 if (D > (L2 + L3) || D < fabsf(L2 - L3))
65 {
66 printf("Leg %d IK failed - Distance %.2f out of range [%.2f, %.2f]\n",
67 leg_number, D, fabsf(L2 - L3), L2 + L3);
68 printf(" Target: x=%.2f, y=%.2f, z=%.2f\n", x, y, z);
69 printf(" Local: x=%.2f, y=%.2f\n", local_x, local_y);
70 printf(" r=%.2f, h=%.2f\n", r, h);
71 return false;
72 }
73
74 // 5. Obliczenie gamma (kąt między L2 i L3)
75 float cos_gamma = (D2 - L2 * L2 - L3 * L3) / (2.0f * L2 * L3);
76 cos_gamma = fmaxf(-1.0f, fminf(1.0f, cos_gamma));
77 float gamma = acosf(cos_gamma);
78
79 // 6. Obliczenie kąta kolana (q2)
80 float alpha = atan2f(h, r);
81 float beta = acosf((D2 + L2 * L2 - L3 * L3) / (2.0f * L2 * D));
82 *q2 = -(alpha - beta);
83
84 // 7. Obliczenie kąta kostki (q3)
85 if (leg->invert_knee)
86 {
87 // Dla prawej strony (2,4,6)
88 *q3 = gamma - M_PI;
89 }
90 else
91 {
92 // Dla lewej strony (1,3,5)
93 *q3 = -(M_PI - gamma);
94 }
95
96 printf("Leg %d final angles [deg]: hip=%.1f, knee=%.1f, ankle=%.1f\n",
97 leg_number, *q1 * 180.0f / M_PI, *q2 * 180.0f / M_PI, *q3 * 180.0f / M_PI);
98
99 return true;
100}

103bool debugLegIK(int leg_number, float x, float y, float z)
104{
105 const LegOrigin_t *leg = &leg_origins[leg_number - 1];

107 printf("=== DEBUG IK dla nogi %d ===\n", leg_number);

109 printf("Origin nogi: x=%.3f, y=%.3f\n", leg->x, leg->y);

111 leg->invert_hip ? "true" : "false",
112 leg->invert_knee ? "true" : "false");
113
114 // Lokalne współrzędne
115 float local_x = x - leg->x;
116 float local_y = y - leg->y;
117 printf("Lokalne: x=%.2f, y=%.2f\n", local_x, local_y);
118
119 // Odległość radialna
120 float r = sqrtf(local_x * local_x + local_y * local_y) - L1;
121 float h = -z;
122 float D = sqrtf(r * r + h * h);
123
124 printf("r=%.2f, h=%.2f, D=%.2f\n", r, h, D);
125 printf("Zasięg: min=%.2f, max=%.2f\n", fabsf(L2 - L3), L2 + L3);
126 printf("Długości segmentów: L1=%.1f, L2=%.1f, L3=%.1f\n", L1, L2, L3);
127
128 if (D > (L2 + L3))
129 {
130 printf("Cel za daleko! D=%.2f > max=%.2f (różnica: %.2f)\n",
131 D, L2 + L3, D - (L2 + L3));
132 return false;
133 }
134
135 if (D < fabsf(L2 - L3))
136 {
137 printf("Cel za blisko! D=%.2f < min=%.2f\n", D, fabsf(L2 - L3));
138 return false;
139 }
140
141 printf("IK wykonalne - cel w zasięgu\n");
142
143 // Przetestuj rzeczywiste IK
144 float q1, q2, q3;
145 bool ik_result = computeLegIK(leg_number, x, y, z, &q1, &q2, &q3);
146 printf("Rezultat IK: %s\n", ik_result ? "SUCCESS" : "FAILED");
147
148 if (ik_result)
149 {
150 printf("Kąty [deg]: hip=%.1f, knee=%.1f, ankle=%.1f\n",
151 q1 * 180.0f / M_PI, q2 * 180.0f / M_PI, q3 * 180.0f / M_PI);
152 }
153
154 return ik_result;
155}
