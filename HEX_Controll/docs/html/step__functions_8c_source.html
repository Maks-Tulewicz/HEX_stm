17static float cubicInterpolation(float t)
18{
19 // Ograniczenie do [0,1]
20 if (t <= 0.0f)
21 return 0.0f;
22 if (t >= 1.0f)
23 return 1.0f;
24
25 // Smooth step: 3*t^2 - 2*t^3
26 return t * t * (3.0f - 2.0f * t);
27}

32static float lerp(float start, float end, float t)
33{
34 return start + (end - start) * t;

72
73 // Ustaw pozycje bazowe na podstawie numeru nogi
74 switch (leg_number)
75 {
76 case 1:
77 base_x = 18.0f;
78 base_y = -15.0f;

81 base_x = -18.0f;

83 break; // Prawa przednia

85 base_x = 22.0f;

97 base_x = -18.0f;
98 base_y = 15.0f;
99 break; // Prawa tylna
100 default:
101 printf("Nieprawidłowy numer nogi!\n");
102 return false;
103 }
104
105 printf("Pozycja bazowa: (%.1f, %.1f, %.1f)\n", base_x, base_y, base_z);
106
107 // Pozycja docelowa (krok do przodu)
108 float target_x = base_x;
109 float target_y = base_y - step_length; // Krok do przodu (y+)
110 float target_z = base_z;
111
112 printf("Pozycja docelowa: (%.1f, %.1f, %.1f)\n", target_x, target_y, target_z);
113
114 // Sprawdź czy obie pozycje są osiągalne

118 printf("❌ Pozycja bazowa nieosiągalna!\n");
119 return false;
120 }
121
122 if (!computeLegIK(leg_number, target_x, target_y, target_z, &q1, &q2, &q3))
123 {
124 printf("❌ Pozycja docelowa nieosiągalna!\n");
125 return false;
126 }
127
128 printf("✅ Obie pozycje osiągalne, rozpoczynam krok...\n\n");
129
130 // Czas między punktami
131 uint32_t point_delay = step_duration_ms / num_points;
132
133 // FAZA 1: Krok (swing phase) - 60% czasu
134 int swing_points = (int)(num_points * 0.6f);
135 printf("--- FAZA SWING (%d punktów) ---\n", swing_points);
136
137 for (int i = 0; i <= swing_points; i++)
138 {
139 float t = (float)i / (float)swing_points; // Parametr czasu [0,1]
140 float smooth_t = cubicInterpolation(t); // Smooth interpolacja
141
142 // Interpolacja pozycji X,Y
143 float current_x = lerp(base_x, target_x, smooth_t);
144 float current_y = lerp(base_y, target_y, smooth_t);
145
146 // Trajektoria Z - łuk (podniesienie w środku)
147 float arc_height = 4.0f * lift_height * t * (1.0f - t); // Parabola
148 float current_z = base_z - arc_height; // W górę (mniej ujemne Z)
149
150 printf("Punkt %d/%d: t=%.2f, pos=(%.1f, %.1f, %.1f), lift=%.1f\n",
151 i, swing_points, t, current_x, current_y, current_z, arc_height);
152
153 // Oblicz IK i ustaw serwa
154 if (computeLegIK(leg_number, current_x, current_y, current_z, &q1, &q2, &q3))
155 {
156 setLegJoints(pca, leg_number, q1, q2, q3);
157 }
158 else
159 {
160 printf("❌ IK failed w punkcie %d\n", i);
161 }
162
163 HAL_Delay(point_delay);
164 }
165
166 // FAZA 2: Powrót (stance phase) - 40% czasu
167 int stance_points = num_points - swing_points;
168 printf("\n--- FAZA STANCE (%d punktów) ---\n", stance_points);
169
170 for (int i = 0; i <= stance_points; i++)
171 {
172 float t = (float)i / (float)stance_points;
173 float smooth_t = cubicInterpolation(t);
174
175 // Powrót po ziemi z docelowej do bazowej
176 float current_x = lerp(target_x, base_x, smooth_t);
177 float current_y = lerp(target_y, base_y, smooth_t);
178 float current_z = target_z; // Pozostaje na ziemi
179
180 printf("Punkt %d/%d: t=%.2f, pos=(%.1f, %.1f, %.1f)\n",
181 i, stance_points, t, current_x, current_y, current_z);
182
183 if (computeLegIK(leg_number, current_x, current_y, current_z, &q1, &q2, &q3))
184 {
185 setLegJoints(pca, leg_number, q1, q2, q3);
186 }
187 else
188 {
189 printf("❌ IK failed w fazie stance %d\n", i);
190 }
191
192 HAL_Delay(point_delay);
193 }
194
195 printf("\n✅ Krok zakończony! Noga powróciła do pozycji bazowej.\n");
196 return true;
197}
