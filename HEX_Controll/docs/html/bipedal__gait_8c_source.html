69static float cubicInterpolation(float t)
70{
71 if (t <= 0.0f)
72 return 0.0f;
73 if (t >= 1.0f)
74 return 1.0f;
75 return t * t * (3.0f - 2.0f * t);
76}

102static void setLegJointsWithOffset(int leg_number, float q1, float q2, float q3,
103 PCA9685_Handle_t *pca1, PCA9685_Handle_t *pca2)
104{
105 if (leg_number < 1 || leg_number > 6)
106 return;
107
108 const LegMapping_t *mapping = &leg_mapping[leg_number - 1];
109 PCA9685_Handle_t *pca_to_use = mapping->is_left_side ? pca1 : pca2;

117 // Konwersja z offsetem

119 float knee_deg = q2 * 180.0f / M_PI;

128
129 // Ograniczenia

131 servo_hip = 0.0f;
132 if (servo_hip > 180.0f)
133 servo_hip = 180.0f;
134 if (servo_knee < 0.0f)
135 servo_knee = 0.0f;
136 if (servo_knee > 180.0f)
137 servo_knee = 180.0f;
138 if (servo_ankle < 0.0f)

140 if (servo_ankle > 180.0f)
141 servo_ankle = 180.0f;
142
143 // Ustaw serwa
144 PCA9685_SetServoAngle(pca_to_use, mapping->base_channel + 0, servo_hip);
145 PCA9685_SetServoAngle(pca_to_use, mapping->base_channel + 1, servo_knee);
146 PCA9685_SetServoAngle(pca_to_use, mapping->base_channel + 2, servo_ankle);

152static bool executeSwingPhase(int pair_index, BipedalDirection_t direction,
153 PCA9685_Handle_t *pca1, PCA9685_Handle_t *pca2)
154{
155 printf("\n--- FAZA SWING: Para %d ---\n", pair_index);
156
157 int swing_leg1 = leg_pairs[pair_index][0];
158 int swing_leg2 = leg_pairs[pair_index][1];
159
160 printf("SWING: nogi %d,%d | POZOSTA≈ÅE: stojƒÖ bez ruchu\n", swing_leg1, swing_leg2);
161
162 // ADAPTACYJNY DELAY - mo≈ºe byƒá 0ms dla maksymalnej prƒôdko≈õci
163 uint32_t step_delay = bipedal_config.step_duration_ms / bipedal_config.step_points;
164 // NIE wymuszaj minimum 1ms - pozw√≥l na 0ms dla ultra prƒôdko≈õci
165
166 printf("Swing delay: %lu ms/punkt (total: %d punkt√≥w = %lu ms)\n",
167 step_delay, bipedal_config.step_points, step_delay * bipedal_config.step_points);
168
169 // === FAZA SWING ===
170 for (int i = 0; i <= bipedal_config.step_points; i++)
171 {
172 float t = (float)i / (float)bipedal_config.step_points;
173 float smooth_t = cubicInterpolation(t);
174
175 // SWING dla pary
176 for (int p = 0; p < 2; p++)
177 {
178 int leg_number = (p == 0) ? swing_leg1 : swing_leg2;
179 int leg_index = leg_number - 1;
180
181 float base_x = base_positions[leg_index][0];
182 float base_y = base_positions[leg_index][1];
183 float base_z = base_positions[leg_index][2];
184
185 // Swing: z obecnej pozycji do pozycji przedniej
186 float swing_start_y = leg_current_y[leg_index]; // Obecna pozycja
187 float swing_end_y = base_y - bipedal_config.step_length; // Pozycja przednia
188
189 float current_x = base_x;
190 float current_y = lerp(swing_start_y, swing_end_y, smooth_t);
191
192 // Trajektoria ≈Çuku
193 float arc_height = 4.0f * bipedal_config.lift_height * t * (1.0f - t);
194 float current_z = base_z - arc_height;
195
196 // Oblicz IK i ustaw serwa
197 float q1, q2, q3;
198 if (computeLegIK(leg_number, current_x, current_y, current_z, &q1, &q2, &q3))
199 {
200 setLegJointsWithOffset(leg_number, q1, q2, q3, pca1, pca2);
201 }
202
203 // Zapisz pozycjƒô ko≈ÑcowƒÖ swing
204 if (i == bipedal_config.step_points)
205 {
206 leg_current_y[leg_index] = swing_end_y;
207 printf("SWING KONIEC Noga %d: y=%.1f\n", leg_number, swing_end_y);
208 }
209 }
210
211 // POZOSTA≈ÅE NOGI: stojƒÖ w obecnych pozycjach (bez ruchu)
212 for (int leg = 1; leg <= 6; leg++)
213 {
214 if (leg == swing_leg1 || leg == swing_leg2)
215 continue; // Pomi≈Ñ swing legs
216
217 int leg_index = leg - 1;
218 float current_x = base_positions[leg_index][0];
219 float current_y = leg_current_y[leg_index]; // Obecna pozycja (bez zmian)
220 float current_z = base_positions[leg_index][2];
221
222 float q1, q2, q3;
223 if (computeLegIK(leg, current_x, current_y, current_z, &q1, &q2, &q3))
224 {
225 setLegJointsWithOffset(leg, q1, q2, q3, pca1, pca2);
226 }
227 }
228
229 // USU≈É HAL_Delay dla maksymalnej prƒôdko≈õci!
230 // HAL_Delay(step_delay); // ‚Üê WY≈ÅƒÑCZONE!
231 }
232
233 return true;
234}

239static bool executeStanceShift(BipedalDirection_t direction,
240 PCA9685_Handle_t *pca1, PCA9685_Handle_t *pca2)
241{
242 printf("\n--- FAZA STANCE: Wszystkie nogi przesuwajƒÖ siƒô o 1/3 do ty≈Çu ---\n");
243
244 // ULTRA SZYBKA STANCE - mniej punkt√≥w, szybszy delay
245 int stance_points = 10; // DRASTYCZNIE MNIEJ punkt√≥w
246 uint32_t stance_delay = 20 / stance_points; // 20ms ca≈Çkowity czas stance (by≈Ço 100ms)
247 // Pozw√≥l na 0ms delay dla ultra prƒôdko≈õci
248
249 printf("Stance delay: %lu ms/punkt (total: %d punkt√≥w = %lu ms)\n",
250 stance_delay, stance_points, stance_delay * stance_points);
251
252 float stance_shift = bipedal_config.step_length / 3.0f;
253
254 // Zapisz pozycje startowe stance
255 float stance_start_y[6];
256 for (int i = 0; i < 6; i++)
257 {
258 stance_start_y[i] = leg_current_y[i];
259 }
260
261 // === FAZA STANCE SHIFT ===
262 for (int i = 0; i <= stance_points; i++)
263 {
264 float t = (float)i / (float)stance_points;
265 float smooth_t = cubicInterpolation(t);
266
267 // WSZYSTKIE NOGI przesuwajƒÖ siƒô o 1/3 do ty≈Çu
268 for (int leg = 1; leg <= 6; leg++)
269 {
270 int leg_index = leg - 1;
271
272 float current_x = base_positions[leg_index][0];
273 float current_z = base_positions[leg_index][2];
274
275 // Stance shift: z obecnej pozycji o 1/3 do ty≈Çu
276 float stance_end_y = stance_start_y[leg_index] + stance_shift;
277 float current_y = lerp(stance_start_y[leg_index], stance_end_y, smooth_t);
278
279 float q1, q2, q3;
280 if (computeLegIK(leg, current_x, current_y, current_z, &q1, &q2, &q3))
281 {
282 setLegJointsWithOffset(leg, q1, q2, q3, pca1, pca2);
283 }
284
285 // Zapisz nowƒÖ pozycjƒô
286 if (i == stance_points)
287 {
288 leg_current_y[leg_index] = stance_end_y;
289 if (leg == 1)
290 printf("STANCE SHIFT: +%.1f do ty≈Çu\n", stance_shift);
291 }
292 }
293
294 // USU≈É HAL_Delay dla maksymalnej prƒôdko≈õci!
295 // HAL_Delay(stance_delay); // ‚Üê WY≈ÅƒÑCZONE!
296 }
297
298 return true;
299}
