63static float cubicInterpolation(float t)

146static bool executeSwingPhase(int leg_number, WaveDirection_t direction,
147 PCA9685_Handle_t *pca1, PCA9685_Handle_t *pca2)
148{
149 printf("\n--- FAZA SWING: Noga %d ---\n", leg_number);
150 printf("SWING: noga %d | POZOSTA≈ÅE: stojƒÖ bez ruchu\n", leg_number);
151
152 uint32_t step_delay = wave_config.step_duration_ms / wave_config.step_points;
153 if (step_delay == 0)
154 step_delay = 1;
155
156 int leg_index = leg_number - 1;
157
158 // === FAZA SWING ===
159 for (int i = 0; i <= wave_config.step_points; i++)
160 {
161 float t = (float)i / (float)wave_config.step_points;
162 float smooth_t = cubicInterpolation(t);
163
164 // SWING dla jednej nogi
165 float base_x = base_positions[leg_index][0];
166 float base_y = base_positions[leg_index][1];
167 float base_z = base_positions[leg_index][2];
168
169 // Swing: z obecnej pozycji do pozycji przedniej
170 float swing_start_y = leg_current_y[leg_index]; // Obecna pozycja
171 float swing_end_y = base_y - wave_config.step_length; // Pozycja przednia
172
173 float current_x = base_x;
174 float current_y = lerp(swing_start_y, swing_end_y, smooth_t);
175
176 // Trajektoria ≈Çuku
177 float arc_height = 4.0f * wave_config.lift_height * t * (1.0f - t);
178 float current_z = base_z - arc_height;
179
180 // Oblicz IK i ustaw serwa dla swing nogi
181 float q1, q2, q3;
182 if (computeLegIK(leg_number, current_x, current_y, current_z, &q1, &q2, &q3))
183 {
184 setLegJointsWithOffset(leg_number, q1, q2, q3, pca1, pca2);
185 }
186
187 // Zapisz pozycjƒô ko≈ÑcowƒÖ swing
188 if (i == wave_config.step_points)
189 {
190 leg_current_y[leg_index] = swing_end_y;
191 printf("SWING KONIEC Noga %d: y=%.1f\n", leg_number, swing_end_y);
192 }
193
194 // POZOSTA≈ÅE NOGI: stojƒÖ w obecnych pozycjach (bez ruchu)
195 for (int leg = 1; leg <= 6; leg++)
196 {
197 if (leg == leg_number)
198 continue; // Pomi≈Ñ swing leg
199
200 int other_leg_index = leg - 1;
201 float other_current_x = base_positions[other_leg_index][0];
202 float other_current_y = leg_current_y[other_leg_index]; // Obecna pozycja (bez zmian)
203 float other_current_z = base_positions[other_leg_index][2];
204
205 if (computeLegIK(leg, other_current_x, other_current_y, other_current_z, &q1, &q2, &q3))
206 {
207 setLegJointsWithOffset(leg, q1, q2, q3, pca1, pca2);
208 }
209 }
210
211 HAL_Delay(step_delay);
212 }
213
214 return true;
215}

220static bool executeStanceShift(WaveDirection_t direction,
221 PCA9685_Handle_t *pca1, PCA9685_Handle_t *pca2)
222{
223 printf("\n--- FAZA STANCE: Wszystkie nogi przesuwajƒÖ siƒô o 1/6 do ty≈Çu ---\n");
224
225 int stance_points = 20; // Mniej punkt√≥w dla stance (szybsze)
226 uint32_t stance_delay = 10 / stance_points; // 10ms ca≈Çkowity czas stance
227 if (stance_delay == 0)
228 stance_delay = 1;
229
230 // KLUCZOWA R√ì≈ªNICA: 1/6 zamiast 1/3 (bo 6 n√≥g zamiast 3 par)
231 float stance_shift = wave_config.step_length / 6.0f;

233 // Zapisz pozycje startowe stance

235 for (int i = 0; i < 6; i++)
236 {
237 stance_start_y[i] = leg_current_y[i];

239
240 // === FAZA STANCE SHIFT ===
241 for (int i = 0; i <= stance_points; i++)

245
246 // WSZYSTKIE NOGI przesuwajƒÖ siƒô o 1/6 do ty≈Çu
247 for (int leg = 1; leg <= 6; leg++)
248 {
249 int leg_index = leg - 1;
250

252 float current_z = base_positions[leg_index][2];
253
254 // Stance shift: z obecnej pozycji o 1/6 do ty≈Çu
255 float stance_end_y = stance_start_y[leg_index] + stance_shift;
256 float current_y = lerp(stance_start_y[leg_index], stance_end_y, smooth_t);
257
258 float q1, q2, q3;
259 if (computeLegIK(leg, current_x, current_y, current_z, &q1, &q2, &q3))
260 {
261 setLegJointsWithOffset(leg, q1, q2, q3, pca1, pca2);
262 }
263
264 // Zapisz nowƒÖ pozycjƒô
265 if (i == stance_points)
266 {
267 leg_current_y[leg_index] = stance_end_y;
268 if (leg == 1)
269 printf("STANCE SHIFT: +%.1f do ty≈Çu (1/6)\n", stance_shift);
270 }
271 }
272
273 HAL_Delay(stance_delay);
274 }
