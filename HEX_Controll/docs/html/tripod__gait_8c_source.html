59static float cubicInterpolation(float t)
60{
61 if (t <= 0.0f)
62 return 0.0f;
63 if (t >= 1.0f)
64 return 1.0f;
65 return t * t * (3.0f - 2.0f * t);
66}

71static float lerp(float start, float end, float t)
72{
73 return start + (end - start) * t;
74}

79static void setLegJointsWithOffset(int leg_number, float q1, float q2, float q3,
80 PCA9685_Handle_t *pca1, PCA9685_Handle_t *pca2)
81{
82 if (leg_number < 1 || leg_number > 6)
83 {
84 printf("❌ Nieprawidłowy numer nogi: %d\n", leg_number);
85 return;
86 }
87
88 // Pobierz mapowanie dla nogi
89 const LegMapping_t *mapping = &leg_mapping[leg_number - 1];
90
91 // Wybierz odpowiednią PCA
92 PCA9685_Handle_t *pca_to_use = mapping->is_left_side ? pca1 : pca2;
93
94 if (pca_to_use == NULL)
95 {
96 printf("⚠️ PCA dla nogi %d niedostępne (kanały %d-%d)\n",
97 leg_number, mapping->base_channel, mapping->base_channel + 2);
98 return;
99 }
100
101 // Konwersja radianów na stopnie z offsetem
102 float hip_deg = (q1 * 180.0f / M_PI) + mapping->hip_offset_deg;
103 float knee_deg = q2 * 180.0f / M_PI;
104 float ankle_deg = q3 * 180.0f / M_PI;
105
106 // USUNIĘTO INWERSJĘ KOLAN - wszystkie nogi mają ten sam kierunek
107
108 // Mapowanie na pozycje serw (90° = neutralne)
109 float servo_hip = 90.0f + hip_deg;
110 float servo_knee = 90.0f + knee_deg;
111 float servo_ankle = 90.0f + ankle_deg;
112
113 // Ograniczenia serw
114 if (servo_hip < 0.0f)
115 servo_hip = 0.0f;
116 if (servo_hip > 180.0f)
117 servo_hip = 180.0f;
118 if (servo_knee < 0.0f)
119 servo_knee = 0.0f;
120 if (servo_knee > 180.0f)
121 servo_knee = 180.0f;
122 if (servo_ankle < 0.0f)
123 servo_ankle = 0.0f;
124 if (servo_ankle > 180.0f)
125 servo_ankle = 180.0f;
126
127 printf("Noga %d [kanały %d-%d]: IK[%.1f°, %.1f°, %.1f°] + offset[%.1f°] -> Servo[%.1f°, %.1f°, %.1f°]\n",
128 leg_number, mapping->base_channel, mapping->base_channel + 2,
129 hip_deg - mapping->hip_offset_deg, knee_deg, ankle_deg, mapping->hip_offset_deg,
130 servo_hip, servo_knee, servo_ankle);
131
132 // Ustaw serwa
133 PCA9685_SetServoAngle(pca_to_use, mapping->base_channel + 0, servo_hip); // Hip
134 PCA9685_SetServoAngle(pca_to_use, mapping->base_channel + 1, servo_knee); // Knee
135 PCA9685_SetServoAngle(pca_to_use, mapping->base_channel + 2, servo_ankle); // Ankle
136}

141static void calculateTargetPosition(int leg_number, TripodDirection_t direction,
142 float *target_x, float *target_y, float *target_z)
143{
144 // Pozycja bazowa
145 float base_x = base_positions[leg_number - 1][0];
146 float base_y = base_positions[leg_number - 1][1];
147 float base_z = base_positions[leg_number - 1][2];
148
149 *target_x = base_x;
150 *target_y = base_y;
151 *target_z = base_z;
152
153 // Modyfikacja na podstawie kierunku
154 switch (direction)
155 {
156 case TRIPOD_FORWARD:
157 *target_y = base_y - tripod_config.step_length; // Do przodu (Y-)
158 break;
159 case TRIPOD_BACKWARD:
160 *target_y = base_y + tripod_config.step_length; // Do tyłu (Y+)
161 break;
162 case TRIPOD_LEFT:
163 *target_x = base_x + tripod_config.step_length; // W lewo (X+)
164 break;
165 case TRIPOD_RIGHT:
166 *target_x = base_x - tripod_config.step_length; // W prawo (X-)
167 break;
168 case TRIPOD_TURN_LEFT:
169 // Obrót w lewo - przednie nogi w lewo, tylne w prawo
170 if (leg_number == 1 || leg_number == 2)
171 {
172 *target_x = base_x + tripod_config.step_length; // Przednie w lewo
173 }
174 else if (leg_number == 5 || leg_number == 6)
175 {
176 *target_x = base_x - tripod_config.step_length; // Tylne w prawo
177 }
178 break;
179 case TRIPOD_TURN_RIGHT:
180 // Obrót w prawo - przednie nogi w prawo, tylne w lewo
181 if (leg_number == 1 || leg_number == 2)
182 {
183 *target_x = base_x - tripod_config.step_length; // Przednie w prawo
184 }
185 else if (leg_number == 5 || leg_number == 6)
186 {
187 *target_x = base_x + tripod_config.step_length; // Tylne w lewo
188 }
189 break;
190 default:
191 break;
192 }
193}

198static void executeSwingPoint(int leg_number, TripodDirection_t direction,
199 float t, float smooth_t, PCA9685_Handle_t *pca1, PCA9685_Handle_t *pca2)
200{
201 float base_x = base_positions[leg_number - 1][0];
202 float base_y = base_positions[leg_number - 1][1];
203 float base_z = base_positions[leg_number - 1][2];
204
205 // Pozycja startowa (tylna)
206 float start_x = base_x;
207 float start_y = base_y;
208
209 switch (direction)
210 {
211 case TRIPOD_FORWARD:
212 start_y = base_y + tripod_config.step_length;
213 break;
214 case TRIPOD_BACKWARD:
215 start_y = base_y - tripod_config.step_length;
216 break;
217 case TRIPOD_LEFT:
218 start_x = base_x - tripod_config.step_length;
219 break;
220 case TRIPOD_RIGHT:
221 start_x = base_x + tripod_config.step_length;
222 break;
223 case TRIPOD_TURN_LEFT:
224 if (leg_number == 1 || leg_number == 2)
225 {
226 start_x = base_x - tripod_config.step_length;
227 }
228 else if (leg_number == 5 || leg_number == 6)
229 {
230 start_x = base_x + tripod_config.step_length;
231 }
232 break;
233 case TRIPOD_TURN_RIGHT:
234 if (leg_number == 1 || leg_number == 2)
235 {
236 start_x = base_x + tripod_config.step_length;
237 }
238 else if (leg_number == 5 || leg_number == 6)
239 {
240 start_x = base_x - tripod_config.step_length;
241 }
242 break;
243 default:
244 break;
245 }
246
247 // Pozycja docelowa (przednia)
248 float target_x, target_y, target_z;
249 calculateTargetPosition(leg_number, direction, &target_x, &target_y, &target_z);
250
251 // Interpolacja pozycji
252 float current_x = lerp(start_x, target_x, smooth_t);
253 float current_y = lerp(start_y, target_y, smooth_t);
254
255 // Trajektoria łuku
256 float arc_height = 4.0f * tripod_config.lift_height * t * (1.0f - t);
257 float current_z = base_z - arc_height;
258
259 // Oblicz IK i ustaw serwa
260 float q1, q2, q3;
261 if (computeLegIK(leg_number, current_x, current_y, current_z, &q1, &q2, &q3))
262 {
263 setLegJointsWithOffset(leg_number, q1, q2, q3, pca1, pca2);
264 }
265}

339bool tripodGaitCycle(PCA9685_Handle_t *pca1, PCA9685_Handle_t *pca2, TripodDirection_t direction)
340{
341 printf("\n=== TRIPOD GAIT CYCLE - ULTRA SZYBKI ===\n");
342 printf("Kierunek: %s\n", (direction == TRIPOD_FORWARD) ? "PRZÓD" : (direction == TRIPOD_BACKWARD) ? "TYŁ"
343 : (direction == TRIPOD_LEFT) ? "LEWO"
344 : (direction == TRIPOD_RIGHT) ? "PRAWO"
345 : (direction == TRIPOD_TURN_LEFT) ? "OBRÓT LEWO"
346 : "OBRÓT PRAWO");
347
348 // DRASTYCZNE ZMNIEJSZENIE PUNKTÓW dla prędkości
349 int fast_points = 30; // Zamiast 120! Wciąż płynnie ale 4x szybciej
350
351 printf("FAST MODE: używam %d punktów zamiast %d/%d\n",
352 fast_points, tripod_config.swing_points, tripod_config.stance_points);
353 printf("I2C1: %s, I2C2: %s\n",
354 (pca1 != NULL) ? "CONNECTED" : "NULL",
355 (pca2 != NULL) ? "CONNECTED" : "NULL");
356
357 // FAZA 1: Grupa A (1,4,5) SWING równocześnie z Grupa B (2,3,6) STANCE
358 printf("\n--- FAZA 1: Grupa A swing + Grupa B stance (FAST) ---\n");
359
360 uint32_t start_time = HAL_GetTick();
361
362 // BEZ DELAY - maksymalna prędkość
363 for (int i = 0; i <= fast_points; i++)
364 {
365 float t = (float)i / (float)fast_points;
366 float smooth_t = cubicInterpolation(t);
367
368 // === GRUPA A - SWING (1,4,5) ===
369 executeSwingPoint(1, direction, t, smooth_t, pca1, pca2);
370 executeSwingPoint(4, direction, t, smooth_t, pca1, pca2);
371 executeSwingPoint(5, direction, t, smooth_t, pca1, pca2);
372
373 // === GRUPA B - STANCE (2,3,6) ===
374 executeStancePoint(2, direction, t, smooth_t, pca1, pca2);
375 executeStancePoint(3, direction, t, smooth_t, pca1, pca2); // TWOJA TESTOWA
376 executeStancePoint(6, direction, t, smooth_t, pca1, pca2);
377
378 // BEZ HAL_Delay() - pure speed!
379 }
380
381 uint32_t phase1_time = HAL_GetTick() - start_time;
382 printf("Faza 1 wykonana w %lu ms\n", phase1_time);
383
384 // FAZA 2: Grupa B (2,3,6) SWING równocześnie z Grupa A (1,4,5) STANCE
385 printf("\n--- FAZA 2: Grupa B swing + Grupa A stance (FAST) ---\n");
386
387 start_time = HAL_GetTick();
388
389 // BEZ DELAY - maksymalna prędkość
390 for (int i = 0; i <= fast_points; i++)
391 {
392 float t = (float)i / (float)fast_points;
393 float smooth_t = cubicInterpolation(t);
394
395 // === GRUPA B - SWING (2,3,6) ===
396 executeSwingPoint(2, direction, t, smooth_t, pca1, pca2);
397 executeSwingPoint(3, direction, t, smooth_t, pca1, pca2); // TWOJA TESTOWA
398 executeSwingPoint(6, direction, t, smooth_t, pca1, pca2);
399
400 // === GRUPA A - STANCE (1,4,5) ===
401 executeStancePoint(1, direction, t, smooth_t, pca1, pca2);
402 executeStancePoint(4, direction, t, smooth_t, pca1, pca2);
403 executeStancePoint(5, direction, t, smooth_t, pca1, pca2);
404
405 // BEZ HAL_Delay() - pure speed!
406 }
407
408 uint32_t phase2_time = HAL_GetTick() - start_time;
409 uint32_t total_time = phase1_time + phase2_time;
410
411 printf("Faza 2 wykonana w %lu ms\n", phase2_time);
412 printf("✅ CAŁY CYKL: %lu ms (target: %lu ms)\n",
413 total_time, tripod_config.swing_duration_ms + tripod_config.stance_duration_ms);
414
415 return true;
416}
